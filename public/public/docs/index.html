    const url = new URL(
    "https://app.babsaa.com/public/connector/api/get-location"
);

let headers = {
    "Content-Type": "application/json",
    "Accept": "application/json",
    "Authorization": "Bearer {token}",
};

let body = {
    "lat": "41.40338",
    "lon": "2.17403"
}

fetch(url, {
    method: "GET",
    headers: headers,
